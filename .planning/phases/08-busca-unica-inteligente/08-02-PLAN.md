```
// score: 0 = igual exato; 1 = digitos iguais ignorando letra/formatacao; 2 = substring (fallback, marca "aproximado")
// retorna null quando NAO casa (equivalente ao false de hoje)
function matchScoreQ(nq, qU){
  if(!qU) return 0; // sem filtro = todo mundo "exato" (nao ha selo)
  if(nq===qU) return 0;
  if(nq.replace(/\D/g,"")===qU.replace(/\D/g,"") && qU.replace(/\D/g,"")) return 1;
  if(nq.includes(qU)) return 2;
  return null;
}
function matchScoreL(nl, lU){
  if(!lU) return 0;
  if(nl===lU) return 0;
  if(nl.replace(/[^0-9\/]/g,"").split("/").includes(lU)) return 1; // "20/21" bate "20" ou "21" por token exato
  if(nl.includes(lU)) return 2;
  return null;
}
// rua: fronteira de palavra em vez de substring cru. Comparar por tokens com padding de espaco
// (a abordagem "sem regex dinamica arriscada" recomendada em SEARCH.md 4b) em vez de RegExp(escapeRegex(...)).
function matchScoreRua(log, rCore, rD){
  if((" "+log+" ").includes(" "+rCore+" ")) return 0;              // fronteira de palavra exata
  if(rD && (log.match(/\d+/g)||[]).includes(rD)) return 1;          // rua numerada, token de digito igual
  if(log.includes(rCore)) return 2;                                  // substring cru (fallback, "aproximado")
  return null;
}
```

Ponto de fronteira de palavra (SEARCH.md 4b): em vez de `RegExp('\\b'+escapeRegex(rCore)+'\\b')` (regex dinamica),

caractere especial no `rCore` quebrar a regex. Cobre "Rua Bela" vs "Rua Bela Vista" (hierarquia esperada, o termo
completo aparece) sem casar "Belantina" por acidente (nao ha " BELA " dentro de " BELANTINA ").
