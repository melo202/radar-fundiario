```js
function detectMode(textoBruto){
  const raw = clean(textoBruto);
  const digitsOnly = raw.replace(/\D/g,"");
  const n = norm(raw);

  // 1. Inscricao — so digitos, comprimento decide o campo (mesma regra de buscar() MODE=insc)
  if(digitsOnly.length===raw.replace(/\s/g,"").length && digitsOnly.length>=10){
    return digitsOnly.length>10
      ? {mode:"insc", field:"nrinscr", value:digitsOnly, confidence:"alta", label:`Inscricao (unidade) - ${digitsOnly.length} dig.`}
      : {mode:"insc", field:"ci",      value:digitsOnly, confidence:"alta", label:`Inscricao (lote) - ${digitsOnly.length} dig.`};
  }

  // 2. Quadra+Lote — regex tolerante a "q128 lote 5", "quadra 128 lt 5", "q.128/5", "128/5"
  const mQL = n.match(/\bQ(?:D|UADRA)?\.?\s*(\d+[A-Z]?)\b.*?\b(?:L(?:T|OTE)?)\.?\s*(\d+(?:\/\d+)?[A-Z]?)\b/)
           || n.match(/\bQ(?:D|UADRA)?\.?\s*(\d+[A-Z]?)\b/);
  if(mQL){
    return {mode:"ql", quadra:mQL[1], lote:mQL[2]||"", confidence:"alta",
            label: mQL[2]?`Quadra ${mQL[1]} - Lote ${mQL[2]}`:`Quadra ${mQL[1]}`};
  }

  // 3. Endereco — tipo de via explicito, NAO ancorado (TIPOVIA_DETECT, derivado de TIPOVIA)
  const mVia = raw.match(TIPOVIA_DETECT);
  if(mVia){
    const numMatch = raw.match(/\b(\d{1,5})\b\s*$/);
    return {mode:"addr", rua: raw, numero: numMatch?numMatch[1]:"", confidence:"alta",
            label:`Endereco - ${raw}`};
  }

  // 4. Predio — prefixo textual tipico
  const mPredio = n.match(/\b(ED|EDF|EDIFICIO|RES|RESIDENCIAL|COND|CONDOMINIO)\.?\s+(.+)/);
  if(mPredio){
    return {mode:"bd", predio: mPredio[2], confidence:"alta", label:`Predio - ${mPredio[2]}`};
  }

  // 5. Numero curto isolado, sem contexto — AMBIGUO
  if(/^\d{1,4}$/.test(digitsOnly) && digitsOnly===raw.replace(/\s/g,"")){
    return {mode:null, confidence:"baixa", raw, label:"Ambiguo", digits:digitsOnly};
  }

  // 6. Texto sem numero nem tipo de via — tenta setor (COMBO); senao, predio com confidence media
  const setorHit = COMBO.filter(b=>b.search.includes(n));
  if(setorHit.length) return {mode:"ql", bairroOnly:true, confidence:"media", label:`Setor - ${setorHit[0].disp}`};
  return {mode:"bd", predio: raw, confidence:"media", label:`Predio (?) - ${raw}`};
}
```

Risco documentado (SEARCH.md secao 7, riscos): `resolveBairro()` autosseleciona o melhor match SEM
score minimo de confianca — bom para o campo de setor dedicado (usuario ja sinalizou intencao), mas arriscado
se aplicado sobre fragmentos de frase livre (mais chance de substring acidental). Para o setor-na-frase,
EXIGIR que o match esteja ANCORADO NO INICIO da frase (nao em qualquer posicao) — mais estrito que o
`resolveBairro()` generico, para reduzir falso-positivo quando a origem e a caixa unica.
