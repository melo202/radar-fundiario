/* SCORE-01 (Fase 9): score de oportunidade 0-100 pela posicao do R$/môý do imovel (myPm2) vs
   mediana/quartis da vizinhanca (stats: {med,q1,q3,n}, mesmo shape de compsStats()/getComps()).
   Formula: percentil aproximado de myPm2 dentro da distribuicao (usa q1/med/q3 como ancoras --
   interpolacao linear em 4 segmentos: [min~q1-1.5iqr, q1], [q1,med], [med,q3], [q3,max~q3+1.5iqr]).
   score = 100 - percentil (abaixo da mediana = mais oportunidade pro comprador, score alto).
   SEM BASE (retorna null, nunca inventa): stats ausente, stats.n<3, ou myPm2 nao finito/<=0.
   flags.atipico (imovel garagem/box/etc, via isGarage) NAO bloqueia o score mas rebaixa o rotulo
   de confianca em scoreConfianca (nao aqui) -- score de oportunidade eh so posicao de preco. */
function scoreOportunidade(myPm2, stats, flags){
  if(!stats || !(stats.n>=3) || !(myPm2>0) || !isFinite(myPm2)) return null;
  const {q1,med,q3} = stats;
  if(!(q1>0 && med>0 && q3>0) || !(q3>=q1)) return null;
  const iqr = q3-q1;
  const min = stats.min!=null ? stats.min : Math.max(1, q1-1.5*iqr);
  const max = stats.max!=null ? stats.max : q3+1.5*iqr;
  const lerp=(v,a,b,pa,pb)=> b===a ? pa : pa+(v-a)/(b-a)*(pb-pa);
  let pct;
  if(myPm2<=min) pct=0;
  else if(myPm2>=max) pct=100;
  else if(myPm2<=q1) pct=lerp(myPm2,min,q1,0,25);
  else if(myPm2<=med) pct=lerp(myPm2,q1,med,25,50);
  else if(myPm2<=q3) pct=lerp(myPm2,med,q3,50,75);
  else pct=lerp(myPm2,q3,max,75,100);
  const score=Math.round(100-pct);
  const diffPct=Math.round(Math.abs(myPm2-med)/med*100);
  const abaixo=myPm2<med;
  let rotulo, porque;
  if(score>=66){
    rotulo="Boa oportunidade";
    porque=[`EstûÀ ${diffPct}% abaixo da mediana da vizinhanûÏa (comparûÀveis em atûˋ ${(flags&&flags.radius)||400} m).`];
  } else if(score>=33){
    rotulo="Oportunidade mûˋdia";
    porque=[abaixo
      ? `EstûÀ ${diffPct}% abaixo da mediana da vizinhanûÏa.`
      : `EstûÀ ${diffPct}% acima da mediana da vizinhanûÏa.`];
  } else {
    rotulo="Abaixo da mediana";
    porque=[`EstûÀ ${diffPct}% acima da mediana da vizinhanûÏa ã entre os mais caros da regiûÈo.`];
  }
  return {score, rotulo, porque};
}

/* SCORE-02 (Fase 9): confianca alta/media/baixa por COMPLETUDE de dados (nunca por qualidade de preco).
   Criterios (cada ausencia = 1 pendencia citada em porque[]):
   - areaOk: area (privativa ou total, conforme o tipo) confirmada/presente (>0)
   - nComps: numero de comparaveis validos na vizinhanca (0 se sem consulta ainda)
   - atipico: imovel atipico na amostra (garagem/box/etc via isGarage)
   - venalOk: venal informado (>0) -- sem venal, a base fiscal do calculo fica mais fraca
   nivel: alta (0 pendencias E nComps>=8), media (<=1 pendencia E nComps>=3), baixa (demais casos) */
function scoreConfianca(inputs){
  const {areaOk,nComps,atipico,venalOk} = inputs||{};
  const n = nComps||0;
  /* (fix plan-check) PENDENCIAS (defeitos contaveis) separadas de NOTAS (informativas):
     n entre 3 e 7 e uma NOTA no porque[], nao um defeito ã senao qualquer 1 defeito real
     + 3..7 comparaveis viraria sempre 2 "pendencias" => "baixa", contradizendo a spec
     (fixture obrigatoria: areaOk=false + nComps=6 => "media"). So n<3 conta como defeito. */
  const pendencias=[];
  const notas=[];
  if(!areaOk) pendencias.push("faltou a ûÀrea confirmada");
  if(!venalOk) pendencias.push("valor venal nûÈo informado");
  if(atipico) pendencias.push("imû°vel atûÙpico na amostra (uso pouco comum na vizinhanûÏa)");
  if(n<3) pendencias.push(`poucos comparûÀveis na vizinhanûÏa (${n})`);
  else if(n<8) notas.push(`${n} comparûÀveis na vizinhanûÏa`);
  let nivel;
  if(pendencias.length===0 && n>=8) nivel="alta";
  else if(pendencias.length<=1 && n>=3) nivel="media";
  else nivel="baixa";
  const partes = pendencias.concat(notas);
  const porque = partes.length
    ? [partes.join("; ")+"."]
    : [`${n} comparûÀveis na vizinhanûÏa, dados completos.`];
  return {nivel, porque};
}

/* LEIT-01 (Fase 9): leitura pratica em pt-BR por TEMPLATE determinûÙstico (zero LLM, zero jargao --
   mediana/percentil/quartil NUNCA aparecem aqui, so em "ver metodologia"). Combina rotulo de uso
   (residencial/comercial/etc, ja mapeado por USO[]) + bairro + resultado de scoreOportunidade/
   scoreConfianca (podem ser null). Templates enumerados, sem geracao livre de texto. */
function leituraPratica(inputs){
  const {tipoImovel,bairro,oportunidade,confianca} = inputs||{};
  const local = bairro ? `no ${bairro}` : "nesta regiûÈo";
  const sujeito = tipoImovel || "Imû°vel";
  if(!oportunidade){
    return "Dados insuficientes para uma leitura de mercado ã confira os dados tûˋcnicos abaixo.";
  }
  const liquidezBaixa = confianca && confianca.nivel==="baixa";
  if(oportunidade.score>=66){
    return `${sujeito} ${local}. Boa liquidez esperada` +
      (liquidezBaixa ? ", se ûÀrea e conservaûÏûÈo forem confirmadas." : " ã preûÏo competitivo para a regiûÈo.");
  }
  if(oportunidade.score>=33){
    return `${sujeito} ${local}. Imû°vel alinhado û  regiûÈo` +
      (liquidezBaixa ? ", com boa liquidez se ûÀrea privativa e conservaûÏûÈo forem confirmadas." : ".");
  }
  return `${sujeito} ${local}. PreûÏo acima da mûˋdia da vizinhanûÏa ã negociaûÏûÈo pode ser necessûÀria para boa liquidez.`;
}
