```js
// Progressive enhancement: animate ONLY when Motion loaded AND motion not reduced.
// Returns the Motion animation controls (with .stop()) or null. NEVER throws.
// Callers MUST perform the underlying state change (data-view swap, .show toggle, DOM insert)
// UNCONDITIONALLY, outside this call — app function must be 100% independent of Motion.
function mAnimate(el, keyframes, options){
  if (REDUCE || !window.Motion || typeof Motion.animate !== "function") return null;
  try { return Motion.animate(el, keyframes, options); } catch(e){ return null; }
}
```
