function showBreadcrumb(nm){
  const bc=document.getElementById("breadcrumb");
  const lbl=(nm&&nm.trim())?esc(nm):"Área rural / gleba";
  document.getElementById("crumbBairro").textContent=lbl; /* textContent = zero risco de injeção */

function hideBreadcrumb(){ const bc=document.getElementById("breadcrumb"); if(bc)bc.style.display="none"; }

}
